#include<Cstring>
#include<iostream> 

using namespace std;

char *search( char *s, char *t ){
  int i, j, k=0, slen, tlen;
  char *p=NULL;
  slen = strlen(s);
  tlen = strlen(t);

  for (i=0; i<slen; i++){
    j=i;
    while(s[j]==t[k]){
      k++;
      j++;
    }
    if(k>=tlen){
      p = &s[i];
      return p;
    }
    k=0;
  }
  return p;
}

int main(){
    int MAXS =30; 
    char s[MAXS], t[MAXS], *pos;

    cin >>s;
    cin >> t; 
    pos = search(s, t);
    if ( pos != NULL )
        cout<< (pos - s);
    else
        cout << "-1\n";

    return 0;
}
